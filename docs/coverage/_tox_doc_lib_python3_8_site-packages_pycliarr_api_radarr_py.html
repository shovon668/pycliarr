63 def get_movie(self, movie_id: Optional[int] = None) -> Union[RadarrMovieItem, List[RadarrMovieItem]]: 

64 """Get specified movie, or all if no id provided from server collection.

66 Args:

71 res = self.get_item(movie_id)

77 def lookup_movie( 

78 self, term: Optional[str] = None, imdb_id: Optional[str] = None, tmdb_id: Optional[int] = None 

79 ) -> Optional[Union[RadarrMovieItem, List[RadarrMovieItem]]]: 

80 """Search for a movie based on keyword, or imbd/tmdb id. 

81 

82 If no imdb id is provided, tvdb id will be used. If neither of them is provided, the keyword will be used. 

83 One of ``term``, ``imdb_id``, or ``tmdb_id`` must be specified. 

84 

85 Args: 

86 term (Optional[str]): Keywords to seach for 

87 imdb_id (Optional[str]): IMDB movie id 

88 tmdb_id (Optional[int]): TMDB movie id 

89 Returns: 

90 json response 

91 """ 

92 if tmdb_id: 

93 url_path = f"{self.api_url_itemlookup}/tmdb" 

94 url_params: Dict[str, Any] = {"tmdbId": tmdb_id} 

95 res = self.request_get(url_path, url_params=url_params) 

96 elif imdb_id: 

97 url_path = f"{self.api_url_itemlookup}/imdb" 

98 url_params = {"imdbId": imdb_id} 

99 res = self.request_get(url_path, url_params=url_params) 

100 elif term: 

101 res = self.lookup_item(str(term)) 

102 else: 

103 raise RadarrCliError("Error, invalid parameters") 

104 

105 if not res: 

106 return None 

107 elif isinstance(res, list): 

108 return [RadarrMovieItem.from_dict(serie) for serie in res] 

109 else: 

110 return RadarrMovieItem.from_dict(res)

112 def add_movie( 

113 self, 

114 quality: int, 

115 tmdb_id: Optional[int] = None, 

116 imdb_id: Optional[str] = None, 

117 movie_info: Optional[RadarrMovieItem] = None, 

118 monitored: bool = True, 

119 search: bool = True, 

120 ) -> json_data: 

121 """addMovie adds a new movie to collection. 

122 

123 The movie description movie_info must be specified. If the IMDB or TMDB id is provided instead, 

124 it will be used to fetch the required movie description from TMDB. 

125 

126 Args: 

127 quality: Quality profile to use, as retrieved by get_quality_profiles() 

128 imdb_id (Optional[int]): IMDB id of the movie to add 

129 tmdb_id (Optional[int]): TMDB id of the movie to add 

130 movie_info (Optional[RadarrMovieItem]): Description of the movie to add 

131 monitored (bool): Whether to monitor the movie. Default is True 

132 search (bool): Whether to search for the movie once added. Default is True 

133 Returns: 

134 json response 

135 """ 

136 

137 # Get info from imdb/tmdb if needed: 

138 if tmdb_id or imdb_id: 

139 movie_info = cast(RadarrMovieItem, self.lookup_movie(tmdb_id=tmdb_id, imdb_id=imdb_id)) 

140 if not movie_info: 

141 raise RadarrCliError("Error, invalid parameters or invalid tmdb/imdb id") 

142 

143 # Prepare movie info for adding 

144 root_path = self.get_root_folder() 

145 movie_info.path = root_path["path"] + movie_info.title 

146 movie_info.profileId = quality 

147 movie_info.qualityProfileId = quality 

148 movie_info.monitored = monitored 

149 movie_info.add_attribute("addOptions", {"searchForMovie": search}) 

150 

151 return self.add_item(json_data=movie_info.to_dict())
