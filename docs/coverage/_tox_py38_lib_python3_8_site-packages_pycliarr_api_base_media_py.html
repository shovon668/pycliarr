31 def get_calendar(self, start_date: Optional[datetime] = None, end_date: Optional[datetime] = None) -> json_dict: 

32 """Retrieve info about when items were/will be downloaded. 

33 

34 If start and end are not provided, retrieves movies airing today and tomorrow. 

35 Args: 

36 start_date (Optional[datetime]): Start date of events to retrieve 

37 end_date (Optional[datetime]): End date of events to retrieve 

38 Returns: 

39 json response 

40 """ 

41 url_params = {} 

42 if start_date and end_date: 

43 url_params["start"] = start_date.strftime("%Y-%m-%d") 

44 url_params["end"] = end_date.strftime("%Y-%m-%d") 

45 return self.request_get(self.api_url_calendar, url_params=url_params)
